51RdaFile parse_rda(Reader_& reader, const ParseRdaOptions& options) {
52 std::unique_ptr<byteme::BufferedReader<unsigned char> > srcptr;
53 if (options.parallel) {
54 srcptr.reset(new byteme::SerialBufferedReader<unsigned char, Reader_*>(&reader, options.buffer_size));
55 } else {
56 srcptr.reset(new byteme::ParallelBufferedReader<unsigned char, Reader_*>(&reader, options.buffer_size));
57 }
58 auto& src = *srcptr;
59
60 RdaFile output;
61
62 // Reading the header first. This is the first and only time that
63 // we need to do a src.valid() check, as we're using the current
64 // position of the source; in all other cases, it can be assumed
65 // that the source needs to be advance()'d before get().
66 {
67 try {
68 if (!src.valid()) {
69 throw empty_error();
70 }
71
72 std::string header;
73 header += as_char(src.get());
74 for (int i = 0; i < 4; ++i) {
75 if (!src.advance()) {
76 throw empty_error();
77 }
78 header += as_char(src.get());
79 }
80 if (header != "RDX2\n" && header != "RDX3\n") {
81 throw std::runtime_error("unsupported format are currently supported");
82 }
83
84 header.clear();
85 for (int i = 0; i < 2; ++i) {
86 if (!src.advance()) {
87 throw empty_error();
88 }
89 header += as_char(src.get());
90 }
91 if (header != "X\n") {
92 throw std::runtime_error("only RDA files in XDR format are currently supported");
93 }
94 } catch (std::exception& e) {
95 throw traceback("failed to read the header from the RDA preamble", e);
96 }
97
98 output.format_version = 0;
99 try {
100 output.format_version = quick_integer<I<decltype(output.format_version)> >(src);
101 } catch (std::exception& e) {
102 throw traceback("failed to read the format version number from the RDA preamble", e);
103 }
104
105 try {
106 output.writer_version = parse_version(src);
107 } catch (std::exception& e) {
108 throw traceback("failed to read the writer version number from the RDA preamble", e);
109 }
110
111 try {
112 output.reader_version = parse_version(src);
113 } catch (std::exception& e) {
114 throw traceback("failed to read the reader version number from the RDA preamble", e);
115 }
116 }
117
118 // Reading this undocumented section about the string encoding.
119 {
120 std::int32_t encoding_length = 0;
121 try {
122 encoding_length = quick_integer<I<decltype(encoding_length)> >(src);
123 if (encoding_length < 0) {
124 throw std::runtime_error("encoding length should be non-negative");
125 }
126 } catch (std::exception& e) {
127 throw traceback("failed to read the encoding length from the RDA preamble", e);
128 }
129
130 try {
131 std::string encoding;
132 encoding.reserve(encoding_length); // don't resize and use extract() on string::data, as that pointer is read-only AFAICT.
133 for (I<decltype(encoding_length)> b = 0; b < encoding_length; ++b) {
134 if (!src.advance()) {
135 throw empty_error();
136 }
137 encoding.push_back(as_char(src.get()));
138 }
139 output.encoding = string_encoding_from_name(encoding);
140 } catch (std::exception& e) {
141 throw traceback("failed to read the encoding string from the RDA preamble", e);
142 }
143 }
144
145 // Now we can finally read the damn object.
146 SharedParseInfo shared;
147
148 auto details = parse_header(src);
149 auto sexp_type = details[3];
150 if (sexp_type != static_cast<unsigned char>(SEXPType::LIST)) {
151 throw std::runtime_error("expected RDA file to contain a pairlist");
152 }
153
154 auto payload = parse_pairlist_body(src, details, shared);
155 output.objects.reserve(payload->data.size());
156 for (auto& x : payload->data) {
157 if (!x.tag.has_value()) {
158 throw std::runtime_error("expected RDA file to contain a tagged pairlist");
159 }
160 output.objects.emplace_back(std::move(*(x.tag)), std::move(x.value));
161 }
162
163 output.environments = std::move(shared.environments);
164 output.symbols = std::move(shared.symbols);
165 output.external_pointers = std::move(shared.external_pointers);
166
167 return output;
168}
